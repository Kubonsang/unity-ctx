# v0.1 Read-Only MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first usable `unity-ctx` binary that reads Unity YAML fixtures and serves deterministic `summarize`, `query`, `inspect`, and `get` commands for `scene`, `prefab`, and `asset`.

**Architecture:** Use one small Go module with a shared Unity YAML parser, a document/query layer, and a thin CLI layer that maps `<namespace> <command> <file>` requests into read-only operations. Keep parsing conservative: preserve unknown blocks, normalize a few common scalar/object shapes, and return explicit `WARN`/`ERROR` strings instead of guessing.

**Tech Stack:** Go 1.22+, standard library only, `testing`, `encoding/json`, `flag`, `os/exec` for CLI tests

---

## Scope Decision

This plan intentionally follows `AGENTS.md` and the repo "Current Focus" over the broader Rev 5 SRS. That means:

- Include now: parser, `scene|prefab|asset`, `summarize`, `query`, `inspect`, `get`, `--view tiny|compact|detail`, deterministic `--json` where easy.
- Exclude now: `index`, `context-pack`, mutation, backups, bench, Editor integration.

```go
package parser

import (
	"bufio"
	"bytes"
	"fmt"
	"regexp"
	"strconv"
	"strings"
)

var headerRE = regexp.MustCompile(`^--- !u!(\d+) &(\d+)$`)

type Block struct {
	ClassID  int
	FileID   int64
	TypeName string
	Fields   map[string]any
}

type Document struct {
	Blocks []Block
}

func Parse(src []byte) (*Document, error) {
	scanner := bufio.NewScanner(bytes.NewReader(src))
	doc := &Document{}
	var current *Block

	for scanner.Scan() {
		line := scanner.Text()
		if m := headerRE.FindStringSubmatch(line); m != nil {
			classID, _ := strconv.Atoi(m[1])
			fileID, _ := strconv.ParseInt(m[2], 10, 64)
			doc.Blocks = append(doc.Blocks, Block{ClassID: classID, FileID: fileID, Fields: map[string]any{}})
			current = &doc.Blocks[len(doc.Blocks)-1]
			continue
		}
		if current == nil || strings.TrimSpace(line) == "" || strings.HasPrefix(line, "%") {
			continue
		}
		if !strings.HasPrefix(line, "  ") && strings.HasSuffix(line, ":") {
			current.TypeName = strings.TrimSuffix(strings.TrimSpace(line), ":")
			continue
		}
		parseFieldLine(current.Fields, strings.TrimSpace(line))
	}

	if err := scanner.Err(); err != nil {
		return nil, fmt.Errorf("scan: %w", err)
	}
	return doc, nil
}
```

```go
func parseFieldLine(dst map[string]any, line string) {
	parts := strings.SplitN(line, ":", 2)
	if len(parts) != 2 {
		return
	}
	key := strings.TrimSpace(parts[0])
	raw := strings.TrimSpace(parts[1])
	dst[key] = parseValue(raw)
}

func parseValue(raw string) any {
	switch raw {
	case "0":
		return int64(0)
	case "1":
		return int64(1)
	}
	if i, err := strconv.ParseInt(raw, 10, 64); err == nil {
		return i
	}
	if f, err := strconv.ParseFloat(raw, 64); err == nil {
		return f
	}
	if strings.HasPrefix(raw, "{") && strings.HasSuffix(raw, "}") {
		return parseInlineMap(raw)
	}
	return raw
}
```

```go
package document

import (
	"fmt"
	"sort"

	"unity-ctx/internal/parser"
)

type LookupError struct {
	Code    string
	Message string
}

func (e *LookupError) Error() string { return e.Message }

type Doc struct {
	Blocks   []parser.Block
	ByFileID map[int64]int
	ByName   map[string][]int
	ByType   map[string][]int
}

func Build(parsed *parser.Document) *Doc {
	doc := &Doc{
		Blocks:   append([]parser.Block(nil), parsed.Blocks...),
		ByFileID: map[int64]int{},
		ByName:   map[string][]int{},
		ByType:   map[string][]int{},
	}
	for i, block := range doc.Blocks {
		doc.ByFileID[block.FileID] = i
		doc.ByType[block.TypeName] = append(doc.ByType[block.TypeName], i)
		if name, ok := block.Fields["m_Name"].(string); ok && name != "" {
			doc.ByName[name] = append(doc.ByName[name], i)
		}
	}
	for _, ids := range doc.ByName {
		sort.Ints(ids)
	}
	return doc
}
```

func (d *Doc) FindByFileID(fileID int64) (parser.Block, bool) {
	idx, ok := d.ByFileID[fileID]
	if !ok {
		return parser.Block{}, false
	}
	return d.Blocks[idx], true
}

func (d *Doc) FindUniqueByName(name string) (parser.Block, *LookupError) {
	indices := d.ByName[name]
	switch len(indices) {
	case 0:
		return parser.Block{}, &LookupError{Code: "ERROR", Message: fmt.Sprintf("ERROR name not found: %s", name)}
	case 1:
		return d.Blocks[indices[0]], nil
	default:
		return parser.Block{}, &LookupError{Code: "AMBIGUOUS_NAME", Message: fmt.Sprintf("ERROR AMBIGUOUS_NAME name=%s matches=%d", name, len(indices))}
	}
}

func (s *Service) Summarize(namespace, path string, view core.View, jsonOut bool) (core.Result, int) {
	doc, err := s.load(path)
	if err != nil {
		return core.Result{Status: "ERROR", Body: fmt.Sprintf("ERROR %v", err)}, 1
	}

	gameObjects := len(doc.ByType["GameObject"])
	components := len(doc.Blocks) - gameObjects
	unknown := 0
	for _, block := range doc.Blocks {
		if block.ClassID == 0 || block.TypeName == "UnknownComponent" {
			unknown++
		}
	}

	body := fmt.Sprintf("OK %s file=%s game_objects=%d components=%d unknown=%d", strings.ToUpper(namespace), path, gameObjects, components, unknown)
	return core.Result{Status: "OK", Body: body}, 0
}

func (s *Service) Query(namespace, path string, view core.View, jsonOut bool, args QueryArgs) (core.Result, int) {
	doc, err := s.load(path)
	if err != nil {
		return core.Result{Status: "ERROR", Body: fmt.Sprintf("ERROR %v", err)}, 1
	}
	if args.Name != "" {
		block, lookupErr := doc.FindUniqueByName(args.Name)
		if lookupErr != nil {
			return core.Result{Status: "ERROR", Body: lookupErr.Message}, 1
		}
		return core.Result{Status: "FOUND", Body: fmt.Sprintf("FOUND fileID=%d type=%s name=%s", block.FileID, block.TypeName, args.Name)}, 0
	}
	return core.Result{Status: "ERROR", Body: "ERROR query requires --id, --name, or --type"}, 1
}
```

func SortedFieldKeys(fields map[string]any) []string {
	keys := make([]string, 0, len(fields))
	for key := range fields {
		keys = append(keys, key)
	}
	sort.Strings(keys)
	return keys
}

func ResolveField(fields map[string]any, path string) (any, bool) {
	current := any(fields)
	for _, part := range strings.Split(path, ".") {
		obj, ok := current.(map[string]any)
		if !ok {
			return nil, false
		}
		current, ok = obj[part]
		if !ok {
			return nil, false
		}
	}
	return current, true
}

type InspectArgs struct {
	ID        int64
	Name      string
	Component string
}

type GetArgs struct {
	ID        int64
	Name      string
	Component string
	Field     string
}

func (s *Service) Inspect(namespace, path string, view core.View, jsonOut bool, args InspectArgs) (core.Result, int) {
	doc, err := s.load(path)
	if err != nil {
		return core.Result{Status: "ERROR", Body: fmt.Sprintf("ERROR %v", err)}, 1
	}
	block, ok := doc.FindByTypeName(args.Component)
	if !ok {
		return core.Result{Status: "ERROR", Body: fmt.Sprintf("ERROR UNKNOWN_COMPONENT component=%s", args.Component)}, 1
	}
	keys := document.SortedFieldKeys(block.Fields)
	return core.Result{Status: "OK", Body: fmt.Sprintf("OK component=%s fileID=%d fields=%s", block.TypeName, block.FileID, strings.Join(keys, ","))}, 0
}

func (s *Service) Get(namespace, path string, view core.View, jsonOut bool, args GetArgs) (core.Result, int) {
	doc, err := s.load(path)
	if err != nil {
		return core.Result{Status: "ERROR", Body: fmt.Sprintf("ERROR %v", err)}, 1
	}
	block := doc.PrimaryBlockForNamespace(namespace)
	if args.Component != "" {
		var ok bool
		block, ok = doc.FindByTypeName(args.Component)
		if !ok {
			return core.Result{Status: "ERROR", Body: fmt.Sprintf("ERROR UNKNOWN_COMPONENT component=%s", args.Component)}, 1
		}
	}
	value, ok := document.ResolveField(block.Fields, args.Field)
	if !ok {
		return core.Result{Status: "ERROR", Body: fmt.Sprintf("ERROR FIELD_NOT_FOUND field=%s", args.Field)}, 1
	}
	return core.Result{Status: "OK", Body: fmt.Sprintf("OK field=%s value=%v", args.Field, value)}, 0
}
```
