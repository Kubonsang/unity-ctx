# v0.3 Safe Asset Mutation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `asset set` with dry-run-by-default behavior, `--write` gated mutation, `.bak` backup creation, best-effort type hints, and post-write verification without expanding mutation to `scene` or `prefab`.

**Architecture:** Keep the current read-only parser/document/service stack intact and add one narrow mutation layer for `.asset` / `.mat` files only. Mutation should work by reparsing the file, selecting one target block deterministically, rewriting a single scalar field in-place while preserving surrounding YAML, then reparsing again to verify the write.

**Tech Stack:** Go 1.22+, standard library only, existing `internal/parser`, `internal/document`, `internal/app`, `internal/cli`, `testing`, `os`, `path/filepath`, `strconv`, `strings`

## Scope Decision

This plan covers only the smallest coherent v0.3 slice:

- include now: `asset set`, dry-run default, `asset set --write`, `.bak`, best-effort type hints, deterministic output, post-write verification
- exclude now: `scene set`, `prefab set`, `--allow-name-fallback` writes, impact-first flows, multi-field patching

That keeps mutation inside the lowest-risk namespace first and matches the roadmap line item: `set defaults to dry-run`, `asset set --write`, `.bak backup`, `type hints`, `get-based verification`.

## Command Contract For This Slice

Only `asset set` ships in v0.3.

Allowed examples:

```bash
unity-ctx asset set testdata/assets/enemy_config.asset --field maxHealth --value 300
unity-ctx asset set testdata/assets/enemy_config.asset --field maxHealth --value 300 --write
unity-ctx asset set testdata/assets/enemy_config.asset --id 11400000 --field moveSpeed --value 4.0
```

Expected output shape:

```text
DRY_RUN field=maxHealth old=200 new=300 type_hint=int changed=1
WRITE backup=testdata/assets/enemy_config.asset.bak field=maxHealth old=200 new=300 type_hint=int changed=1 verified=1
WARN type_hint=unknown field=customValue old="abc" new="xyz"
```

Expected safety behavior:

- no file changes without `--write`
- `.bak` created before write
- mutation limited to one selected asset block
- non-scalar or missing field returns explicit `ERROR`
- `scene set` and `prefab set` return deterministic “not implemented for namespace” errors

```go
type Block struct {
	ClassID   int
	FileID    int64
	TypeName  string
	Fields    map[string]any
	RawBody   string
	StartLine int
	EndLine   int
}

func Parse(data []byte) ([]Block, error) {
	lines := splitLines(data)
	var blocks []Block

	for i := 0; i < len(lines); {
		line := strings.TrimSpace(lines[i])
		if line == "" || strings.HasPrefix(line, "%") {
			i++
			continue
		}

		match := headerPattern.FindStringSubmatch(line)
		if match == nil {
			return nil, fmt.Errorf("unexpected content outside block header: %q", line)
		}

		// parse classID/fileID as before ...

		j := i + 1
		for j < len(lines) {
			trimmed := strings.TrimSpace(lines[j])
			if headerPattern.MatchString(trimmed) {
				break
			}
			if strings.HasPrefix(trimmed, "---") {
				return nil, fmt.Errorf("unexpected block header: %q", trimmed)
			}
			j++
		}

		bodyLines := lines[i+1 : j]
		block := Block{
			ClassID:   classID,
			FileID:    fileID,
			Fields:    make(map[string]any),
			RawBody:   strings.Join(bodyLines, "\n"),
			StartLine: i,
			EndLine:   j,
		}
		parseBody(bodyLines, &block)
		blocks = append(blocks, block)
		i = j
	}

	return blocks, nil
}
```

```go
package mutation

import (
	"bytes"
	"fmt"
	"os"
	"path/filepath"
	"strconv"
	"strings"

	"unity-ctx/internal/document"
	"unity-ctx/internal/parser"
)

type AssetSetRequest struct {
	HasID bool
	ID    int64
	Field string
	Value string
}

type AssetSetResult struct {
	Field       string
	OldValue    string
	NewValue    string
	TypeHint    string
	Changed     bool
	UpdatedData []byte
}

func PlanAssetSet(input []byte, blocks []parser.Block, req AssetSetRequest) (AssetSetResult, error) {
	doc := document.Build(blocks)
	target, err := resolveAssetTarget(doc, blocks, req)
	if err != nil {
		return AssetSetResult{}, err
	}

	oldValue, ok := document.ResolveField(target.Fields, req.Field)
	if !ok {
		return AssetSetResult{}, fmt.Errorf("FIELD_NOT_FOUND field=%s", req.Field)
	}

	newScalar, typeHint, err := coerceScalar(oldValue, req.Value)
	if err != nil {
		return AssetSetResult{}, err
	}

	updated, changed, err := rewriteScalarField(input, target, req.Field, newScalar)
	if err != nil {
		return AssetSetResult{}, err
	}

	return AssetSetResult{
		Field:       req.Field,
		OldValue:    formatValue(oldValue),
		NewValue:    newScalar,
		TypeHint:    typeHint,
		Changed:     changed,
		UpdatedData: updated,
	}, nil
}

func WriteWithBackup(path string, updated []byte) (string, error) {
	original, err := os.ReadFile(path)
	if err != nil {
		return "", err
	}
	backupPath := path + ".bak"
	if err := os.WriteFile(backupPath, original, 0o644); err != nil {
		return "", err
	}
	if err := os.WriteFile(path, updated, 0o644); err != nil {
		return "", err
	}
	return backupPath, nil
}
```

```go
type SetArgs struct {
	HasID bool
	ID    int64
	Field string
	Value string
	Write bool
}

func (s *Service) Set(namespace, path string, view core.View, jsonOut bool, args SetArgs) (core.Result, int) {
	_ = jsonOut

	result := core.Result{
		Namespace: namespace,
		Command:   "set",
		File:      path,
		View:      view,
	}

	if namespace != "asset" {
		result.Status = "ERROR"
		result.Body = fmt.Sprintf("ERROR set not implemented for namespace=%s", namespace)
		return result, 1
	}
	if strings.TrimSpace(args.Field) == "" {
		result.Status = "ERROR"
		result.Body = "ERROR set requires --field"
		return result, 1
	}
	if strings.TrimSpace(args.Value) == "" {
		result.Status = "ERROR"
		result.Body = "ERROR set requires --value"
		return result, 1
	}

	loaded, err := s.load(path)
	if err != nil {
		result.Status = "ERROR"
		result.Body = fmt.Sprintf("ERROR %v", err)
		return result, 1
	}

	plan, err := mutation.PlanAssetSet(loaded.data, loaded.blocks, mutation.AssetSetRequest{
		HasID: args.HasID,
		ID:    args.ID,
		Field: args.Field,
		Value: args.Value,
	})
	if err != nil {
		result.Status = "ERROR"
		result.Body = "ERROR " + err.Error()
		return result, 1
	}

	if !args.Write {
		result.Status = "OK"
		result.Body = fmt.Sprintf("DRY_RUN field=%s old=%s new=%s type_hint=%s changed=%d",
			plan.Field, plan.OldValue, plan.NewValue, plan.TypeHint, boolToInt(plan.Changed))
		return result, 0
	}

	backupPath, err := mutation.WriteWithBackup(path, plan.UpdatedData)
	if err != nil {
		result.Status = "ERROR"
		result.Body = fmt.Sprintf("ERROR %v", err)
		return result, 1
	}

	verify, verifyCode := s.Get("asset", path, view, false, GetArgs{HasID: args.HasID, ID: args.ID, Field: args.Field})
	if verifyCode != 0 || !strings.Contains(verify.Body, "value="+plan.NewValue) {
		result.Status = "ERROR"
		result.Body = "ERROR VERIFY_FAILED"
		return result, 1
	}

	result.Status = "OK"
	result.Body = fmt.Sprintf("WRITE backup=%s field=%s old=%s new=%s type_hint=%s changed=%d verified=1",
		backupPath, plan.Field, plan.OldValue, plan.NewValue, plan.TypeHint, boolToInt(plan.Changed))
	return result, 0
}
```
