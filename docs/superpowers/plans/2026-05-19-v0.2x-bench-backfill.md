# v0.2x Bench Backfill Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a narrow `bench` command that measures token reduction ratios using the SRS formula `estimated_tokens = ceil(utf8_bytes / 4)`, comparing raw file content against `summarize`, and optionally against `context-pack` only when `--task` is provided.

**Architecture:** Keep token estimation and ratio math in a small pure-Go `internal/bench` package with no tokenizer or Unity Editor dependency. Let `internal/app` orchestrate existing `Summarize` and `ContextPack` flows, and let `internal/cli` expose a compact-only `bench` command with deterministic text and JSON output.

**Tech Stack:** Go standard library, existing `internal/app`, `internal/contextpack`, `internal/core`, existing CLI/service test patterns

## Proposed Public Contract

### Command

```bash
unity-ctx scene bench Assets/Scenes/Stage01.unity
unity-ctx prefab bench Assets/Prefabs/Enemy.prefab --task "inspect combat tuning"
unity-ctx asset bench Assets/Configs/EnemyConfig.asset --json
```

### Flags

- required positional: `<namespace> bench <file>`
- optional:
  - `--task <text>`: when present, also measure `context-pack`
  - `--json`
  - `--view compact` only

### Behavior

- always measure:
  - raw file bytes
  - raw estimated tokens
  - summarize output bytes
  - summarize estimated tokens
  - summarize ratio = `summarize_tokens / raw_tokens`
- only when `--task` is present:
  - run `context-pack` with `Task=args.Task`
  - use a deterministic budget: `max(raw_estimated_tokens, contextpack.MinimumBudgetForOptions(...))`
  - measure context-pack bytes/tokens/ratio
- `bench` itself is read-only
- no external tokenizer
- no timing / latency / CPU / memory measurements

### Compact output

Without `--task`:

```text
OK raw_bytes=512 raw_tokens=128 summarize_bytes=104 summarize_tokens=26 summarize_ratio=0.2031 summarize_saved_tokens=102
```

With `--task`:

```text
OK raw_bytes=512 raw_tokens=128 summarize_bytes=104 summarize_tokens=26 summarize_ratio=0.2031 summarize_saved_tokens=102 context_pack_bytes=80 context_pack_tokens=20 context_pack_ratio=0.1562 context_pack_saved_tokens=108
```

### JSON

Use normal envelope plus nested `bench` payload:

```json
{
  "status": "OK",
  "namespace": "scene",
  "command": "bench",
  "file": "testdata/scenes/simple_scene.unity",
  "view": "compact",
  "body": "OK raw_bytes=...",
  "bench": {
    "raw_bytes": 512,
    "raw_tokens": 128,
    "summarize": {
      "bytes": 104,
      "tokens": 26,
      "ratio": 0.2031,
      "saved_tokens": 102
    },
    "context_pack": {
      "bytes": 80,
      "tokens": 20,
      "ratio": 0.1562,
      "saved_tokens": 108
    }
  }
}
```

`context_pack` is omitted when `--task` is absent.

```go
package bench

type Input struct {
	RawBytes         int
	SummarizeBytes   int
	HasContextPack   bool
	ContextPackBytes int
}

type Metric struct {
	Bytes       int     `json:"bytes"`
	Tokens      int     `json:"tokens"`
	Ratio       float64 `json:"ratio"`
	SavedTokens int     `json:"saved_tokens"`
}

type Result struct {
	RawBytes    int     `json:"raw_bytes"`
	RawTokens   int     `json:"raw_tokens"`
	Summarize   Metric  `json:"summarize"`
	ContextPack *Metric `json:"context_pack,omitempty"`
}

func EstimateTokens(utf8Bytes int) int {
	if utf8Bytes <= 0 {
		return 0
	}
	return (utf8Bytes + 3) / 4
}

func Build(in Input) Result {
	rawTokens := EstimateTokens(in.RawBytes)
	summarizeTokens := EstimateTokens(in.SummarizeBytes)

	out := Result{
		RawBytes:  in.RawBytes,
		RawTokens: rawTokens,
		Summarize: buildMetric(rawTokens, in.SummarizeBytes),
	}

	if in.HasContextPack {
		metric := buildMetric(rawTokens, in.ContextPackBytes)
		out.ContextPack = &metric
	}

	return out
}

func buildMetric(rawTokens, outputBytes int) Metric {
	tokens := EstimateTokens(outputBytes)
	ratio := 0.0
	if rawTokens > 0 {
		ratio = float64(tokens) / float64(rawTokens)
	}
	saved := rawTokens - tokens
	if saved < 0 {
		saved = 0
	}
	return Metric{
		Bytes:       outputBytes,
		Tokens:      tokens,
		Ratio:       ratio,
		SavedTokens: saved,
	}
}
```

Add service method skeleton:

```go
func (s *Service) Bench(namespace, path string, view core.View, jsonOut bool, args BenchArgs) (BenchResult, int) {
	result := BenchResult{
		Result: core.Result{
			Namespace: namespace,
			Command:   "bench",
			File:      path,
			View:      view,
		},
	}

	if view != core.ViewCompact {
		result.Status = "ERROR"
		result.Body = "ERROR bench supports only --view compact"
		return result, 1
	}

	raw, err := os.ReadFile(path)
	if err != nil {
		result.Status = "ERROR"
		result.Body = fmt.Sprintf("ERROR %v", err)
		return result, 1
	}

	summary, code := s.Summarize(namespace, path, core.ViewCompact, false)
	if code != 0 {
		result.Status = summary.Status
		result.Body = summary.Body
		return result, code
	}

	input := bench.Input{
		RawBytes:       len(raw),
		SummarizeBytes: len([]byte(summary.Body)),
	}

	trimmedTask := strings.TrimSpace(args.Task)
	if trimmedTask != "" {
		rawTokens := bench.EstimateTokens(len(raw))
		minBudget := contextpack.MinimumBudgetForOptions(contextpack.Options{
			Namespace: namespace,
			File:      path,
			Task:      trimmedTask,
			MaxTokens: rawTokens,
		}, 0)
		maxBudget := rawTokens
		if maxBudget < minBudget {
			maxBudget = minBudget
		}

		contextResult, contextCode := s.ContextPack(namespace, path, core.ViewCompact, false, ContextPackArgs{
			Task:      trimmedTask,
			MaxTokens: maxBudget,
		})
		if contextCode != 0 {
			result.Status = contextResult.Status
			result.Body = contextResult.Body
			return result, contextCode
		}

		input.HasContextPack = true
		input.ContextPackBytes = len([]byte(contextResult.Body))
	}

	computed := bench.Build(input)
	result.Status = "OK"
	result.Body = formatBenchBody(computed)
	if jsonOut {
		result.Bench = benchPayloadFromResult(computed)
	}
	return result, 0
}
```

```go
func formatBenchBody(result bench.Result) string {
	body := fmt.Sprintf(
		"OK raw_bytes=%d raw_tokens=%d summarize_bytes=%d summarize_tokens=%d summarize_ratio=%.4f summarize_saved_tokens=%d",
		result.RawBytes,
		result.RawTokens,
		result.Summarize.Bytes,
		result.Summarize.Tokens,
		result.Summarize.Ratio,
		result.Summarize.SavedTokens,
	)
	if result.ContextPack != nil {
		body = fmt.Sprintf(
			"%s context_pack_bytes=%d context_pack_tokens=%d context_pack_ratio=%.4f context_pack_saved_tokens=%d",
			body,
			result.ContextPack.Bytes,
			result.ContextPack.Tokens,
			result.ContextPack.Ratio,
			result.ContextPack.SavedTokens,
		)
	}
	return body
}
```
