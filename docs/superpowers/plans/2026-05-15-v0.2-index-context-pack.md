**Goal:** Build the next read-only slice of `unity-ctx`: verifiable `index` snapshots plus token-budgeted `context-pack` output with deterministic `OMITTED` behavior.

**Architecture:** Reuse the current parser/document/service stack and add one new indexing layer plus one focused context-packing layer. Keep `index` as a file-hash-validated snapshot format, and keep `context-pack` intentionally narrow: operate on the existing document model, emit stable compact lines, and degrade with explicit `OMITTED` output instead of guessing.

**Tech Stack:** Go 1.22+, standard library only, existing `testing`, `encoding/json`, `os`, `path/filepath`, `crypto/sha256`

```go
package index

import (
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"fmt"
	"os"

	"unity-ctx/internal/parser"
)

type Snapshot struct {
	SchemaVersion int          `json:"schema_version"`
	Kind          string       `json:"kind"`
	Path          string       `json:"path"`
	FileHash      string       `json:"file_hash"`
	GeneratedBy   string       `json:"generated_by"`
	Objects       []ObjectStub `json:"objects"`
}

type ObjectStub struct {
	FileID   int64  `json:"file_id"`
	ClassID  int    `json:"class_id"`
	TypeName string `json:"type_name"`
	Name     string `json:"name,omitempty"`
}

func BuildSnapshot(kind, path string, blocks []parser.Block) (Snapshot, error) {
	hash, err := FileHash(path)
	if err != nil {
		return Snapshot{}, err
	}

	objects := make([]ObjectStub, 0, len(blocks))
	for _, block := range blocks {
		stub := ObjectStub{
			FileID:   block.FileID,
			ClassID:  block.ClassID,
			TypeName: block.TypeName,
		}
		if name, ok := block.Fields["m_Name"].(string); ok && name != "" {
			stub.Name = name
		}
		objects = append(objects, stub)
	}

	return Snapshot{
		SchemaVersion: 1,
		Kind:          kind,
		Path:          path,
		FileHash:      hash,
		GeneratedBy:   "unity-ctx 0.2.0",
		Objects:       objects,
	}, nil
}

func FileHash(path string) (string, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return "", err
	}
	sum := sha256.Sum256(data)
	return "sha256:" + hex.EncodeToString(sum[:]), nil
}
```

func Save(path string, snapshot Snapshot) error {
	data, err := json.MarshalIndent(snapshot, "", "  ")
	if err != nil {
		return err
	}
	data = append(data, '\n')
	return os.WriteFile(path, data, 0o644)
}

func Load(path string) (Snapshot, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return Snapshot{}, err
	}
	var snapshot Snapshot
	if err := json.Unmarshal(data, &snapshot); err != nil {
		return Snapshot{}, err
	}
	return snapshot, nil
}

func IsStale(snapshot Snapshot, sourcePath string) (bool, string, error) {
	hash, err := FileHash(sourcePath)
	if err != nil {
		return false, "", err
	}
	if hash != snapshot.FileHash {
		return true, "file_hash_mismatch", nil
	}
	return false, "", nil
}
```

package contextpack

import (
	"fmt"

	"unity-ctx/internal/parser"
)

type Options struct {
	Namespace string
	File      string
	Task      string
	Focus     string
	MaxTokens int
}

func EstimateTokens(text string) int {
	n := len([]byte(text))
	return (n + 3) / 4
}

func Build(opts Options, blocks []parser.Block) []string {
	lines := []string{
		fmt.Sprintf("TASK_CONTEXT %s=%s focus=%s budget=%dtok", opts.Namespace, opts.File, opts.Focus, opts.MaxTokens),
	}
	for _, block := range blocks {
		name, _ := block.Fields["m_Name"].(string)
		if name == "" {
			continue
		}
		lines = append(lines, fmt.Sprintf("OBJECT %s id=%d type=%s", name, block.FileID, block.TypeName))
	}
	return enforceBudget(lines, opts.MaxTokens)
}
```

- [ ] **Step 4: Implement deterministic token-budget enforcement**

```go
func enforceBudget(lines []string, maxTokens int) []string {
	if maxTokens <= 0 {
		return lines
	}

	kept := make([]string, 0, len(lines))
	used := 0
	omitted := 0

	for _, line := range lines {
		lineTokens := EstimateTokens(line + "\n")
		if used+lineTokens > maxTokens {
			omitted++
			continue
		}
		kept = append(kept, line)
		used += lineTokens
	}

	if omitted > 0 {
		kept = append(kept, fmt.Sprintf("OMITTED reason=token_budget lines=%d", omitted))
	}
	return kept
}
```
